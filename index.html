真的是很偶然看到这本书的吧，哈哈哈两个网上就看完了，在我这里变成了很合适的睡前图书呀~真的是一本超级可爱的图书《学C编程也可以卡通一点》，就是真的用漫画的形式把语言中大部分内容大概的讲解了一遍~
在里面真的是复习了一遍的赶脚~下面记一下自己回忆起来的一些点吧：
☻double-16字节 int-4字节 char-1字节
☻scanf(“%d %d”,&a,&a)通过地址，将变量的数值放到变量里面
☻编译就是将代码变成程序
☻2==2==1 //true
☻循环中的continue是停止本次循环，直接下一次的循环

☻C语言是典型的顺序编程，而面向对象编程是定义好对象的同时将这个对象应该做的动作全部封装到对象中去。所以当对象建立时，动作自动发生。
☻指针指向某一个变量p=&test;
p将所指的变量数值显示出来
☻数组也是指针；int p=null;p=arr;
☻动态分配变量malloc函数；结束后free函数释放；relloc
之所以产生就是因为不知道要储存多少数据时，采用大数组的话就浪费太多内存了。

发现自己这样一点点记下来其实又看到了更多的东西。乐乐和忧忧的模式其实也可以看成朋友之间相处的过程，从开始到后来的全心接受。也可以理解成，无论是什么样的人都在自己的工作或是生活中寻找一种存在感，价值感。真的真的很重要，这是忧忧在前期“自发性”的去搞一些所谓的“破坏”的原因吧。想到当乐乐跟忧忧说自己要给她指派很重要的任务的时候，忧忧一脸兴奋的样子。

下面记述一下，如何在vba中实现汇总word文章中不同表的内容资料：

` Sub 汇总所有数据()
‘定义所有变量
Dim ff As String, path As String, a As Integer, Filename As String
‘定义等会窗口中过滤出的文件类型
ff = “Word文档(.doc;.docx),.doc;.docx “
‘窗口中左上角显示文字
strtitle = “打开Word文档”
‘选择后取路径，并赋值到path中
path = Application.GetOpenFilename(filefilter:=ff, Title:=strtitle)

‘将“路径”中的路径和文件名整体分开
a = Len(path)
For i = 1 To a
‘从右开始直到截取到\，才执行if程序中内容
If Mid(Right(path, i), 1, 1) = “\” Then
path = Mid(path, 1, a - i + 1)
Exit For
End If
Next
‘按照dir排序path所有文件名
Filename = Dir(path)

'循环查找word文件中table中的指定值，输出到excel中
k = 2
'开始循环，文件不为空，直接开始循环
Do While Filename <> ""
 '下面这一条固定，表示word应用程序（docapp）被创建,同样每一循环都必须打开word
 Set docapp = CreateObject("Word.application")
 '创建对象，打开word文档
 Set dd = docapp.documents.Open(path & Filename)
 '创建对象，附上文档中的第一个表格table
 Set ddd1 = dd.Tables(1)
 Set ddd2 = dd.Tables(2)
 '计算表格中行数
 x = ddd2.Range.Rows.Count
 For i = 2 To x    'i为word中表格的行数
  Cells(k, 1) = Mid(ddd1.cell(1, 2).Range, 1, Len(ddd1.cell(1, 2).Range) - 2)
  Cells(k, 2) = Mid(ddd2.cell(i, 2).Range, 1, Len(ddd2.cell(i, 2).Range) - 2)
  Cells(k, 3) = Mid(ddd1.cell(1, 4).Range, 1, Len(ddd1.cell(1, 4).Range) - 2)
  Cells(k, 4) = Mid(ddd1.cell(1, 6).Range, 1, Len(ddd1.cell(1, 6).Range) - 2)
  Cells(k, 5) = Mid(ddd1.cell(2, 4).Range, 1, Len(ddd1.cell(2, 4).Range) - 2)
  Cells(k, 6) = Mid(ddd1.cell(3, 4).Range, 1, Len(ddd1.cell(3, 4).Range) - 2)
  Cells(k, 7) = Mid(ddd2.cell(i, 3).Range, 1, Len(ddd2.cell(i, 3).Range) - 2)

  k = k + 1      'k为输出到的表格行数
 Next
 '开完一个文档后，各种关闭和归零
 dd.Close
docapp.Quit
 Set docapp = Nothing
 Set dd = Nothing
 Set ddd1 = Nothing
 Set ddd2 = Nothing
 '文件到下一个
 Filename = Dir
 Loop

 '所有文件读取完后，手动修改第一行
 Cells(1, 1) = "姓名"
 Cells(1, 2) = "日期"
 Cells(1, 3) = "性别"
 Cells(1, 4) = "学历"
 Cells(1, 5) = "工资"
 Cells(1, 6) = "年龄"
 Cells(1, 7) = "销售客单"

End Sub `
